def whiten(x):
    mM = np.mean(x,axis=1)
    sph = np.linalg.inv(scp.linalg.sqrtm(np.cov(x)))
    xM = np.subtract(x.T,mM)
    whiteX = np.matmul(sph,xM.T)
    return whiteX
